222 def select(self, items) -> Dict[str, Any]:

227 memory = items.get(OptionalAttribs.MEMORY, "")

229 items["-l select"] = f"={total_nodes}:mpiprocs={cores_per_node}" 

230 if threads_per_core not in NONEISH: 

231 items[ 

232 "-l select" 

233 ] = f"{items['-l select']}:ompthreads={threads_per_core}:ncpus={int(cores_per_node) * int(threads_per_core)}" 

234 if memory not in NONEISH: 

235 items["-l select"] = f"{items['-l select']}:mem={memory}" 

236 return items
