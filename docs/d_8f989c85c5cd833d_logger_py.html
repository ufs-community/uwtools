9class ColoredFormatter(logging.Formatter): 

10 """ 

11 Logging colored formatter 

12 adapted from https://stackoverflow.com/a/56944256/3638629 

13 """ 

14 

15 grey = '\x1b[38;21m' 

16 blue = '\x1b[38;5;39m' 

17 yellow = '\x1b[38;5;226m' 

18 red = '\x1b[38;5;196m' 

19 bold_red = '\x1b[31;1m' 

20 reset = '\x1b[0m'

22 def __init__(self, fmt): 

23 super().__init__() 

24 self.fmt = fmt 

25 self.FORMATS = { 

26 logging.DEBUG: self.blue + self.fmt + self.reset, 

27 logging.INFO: self.grey + self.fmt + self.reset, 

28 logging.WARNING: self.yellow + self.fmt + self.reset, 

29 logging.ERROR: self.red + self.fmt + self.reset, 

30 logging.CRITICAL: self.bold_red + self.fmt + self.reset 

31 } 

32 

33 def format(self, record): 

34 log_fmt = self.FORMATS.get(record.levelno) 

35 formatter = logging.Formatter(log_fmt) 

36 return formatter.format(record)
